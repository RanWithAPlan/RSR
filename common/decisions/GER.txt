GER_executive_council_category = {
    GER_seats_change_mission = {
        activation = { has_country_flag = GER_executive_council_on }
        available = {
            always = no
        }
        priority = 30
        custom_effect_tooltip = GER_seat_change_localisation
        is_good = yes
        days_mission_timeout = 90
        timeout_effect = {  
            add_to_variable = {
                var = GER_anarchisten_seats value = GER_anarchisten_seats_change
            }
            add_to_variable = {
                var = GER_dezentralisten_seats value = GER_dezentralisten_seats_change
            }
            add_to_variable = {
                var = GER_zentralisten_seats value = GER_zentralisten_seats_change
            }
            if = {
                limit = {
                    check_variable = {
                        var = GER_dezentralisten_seats
                        value = GER_anarchisten_seats
                        compare = greater_than_or_equals
                    }
                    check_variable = {
                        var = GER_dezentralisten_seats
                        value = GER_zentralisten_seats
                        compare = greater_than_or_equals
                    }
                }
                set_politics = { ruling_party = council_communism }
                set_country_flag = GER_dezentralisten_election
                clr_country_flag = GER_zentralisten_election
                clr_country_flag = GER_anarchisten_election
            }
            if_else = {
                limit = {
                    check_variable = {
                        var = GER_anarchisten_seats
                        value = GER_dezentralisten_seats
                        compare = greater_than_or_equals
                    }
                    check_variable = {
                        var = GER_anarchisten_seats
                        value = GER_zentralisten_seats
                        compare = greater_than_or_equals
                    }
                }
                set_politics = { ruling_party = anarchism }
                set_country_flag = GER_anarchisten_election
                clr_country_flag = GER_zentralisten_election
                clr_country_flag = GER_dezentralisten_election
            }
            if_else = {
                limit = {
                    check_variable = {
                        var = GER_zentralisten_seats
                        value = GER_dezentralisten_seats
                        compare = greater_than_or_equals
                    }
                    check_variable = {
                        var = GER_zentralisten_seats
                        value = GER_anarchisten_seats
                        compare = greater_than_or_equals
                    }
                }
                set_politics = { ruling_party = radical_socialism }
                set_country_flag = GER_zentralisten_election
                clr_country_flag = GER_anarchisten_election
                clr_country_flag = GER_dezentralisten_election
            }
            activate_mission = GER_seats_change_mission
        }
    }
    #ideology exclusive
    GER_rally_syndicate_support = {
